cmake_minimum_required(VERSION 3.16)
project(AlphabetWasmLsp)

# ---------------------------------------------------------------------------
# WASM-targeted build of the Alphabet Language Server for Node.js hosts.
#
# Produces:
#   build-wasm-lsp/alphabet.wasm       (the compiled LSP server)
#   build-wasm-lsp/alphabet.js         (Emscripten JS glue)
#
# Run:
#   emcmake cmake -S . -B build-wasm-lsp -DCMAKE_BUILD_TYPE=Release
#   cmake --build build-wasm-lsp --target alphabet_lsp
#
# The Node host (editors/vscode-alphabet/server/wasm/lsp-transport.ts) loads
# the resulting alphabet.wasm via WebAssembly.instantiate + WASI, wires
# stdio, and calls `_alphabet_lsp_main` to start the LSP loop.
# ---------------------------------------------------------------------------

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(
    src/include
    ${CMAKE_SOURCE_DIR}/build/generated
)

# Library sources — reuses everything from the main build EXCEPT files
# that pull in popen/system/getenv (vm_builtins.cpp) or other OS deps.
#
# v2.3.5 verification: src/lsp.cpp uses only std::cin/std::cout + std::ifstream.
# It does NOT use fork/pthread/dlopen/popen. WASM compile should be clean.
set(LIB_SOURCES
    src/compiler.cpp
    src/lexer.cpp
    src/parser.cpp
    src/type_system.cpp
    src/lsp.cpp
    src/linter.cpp
    src/bytecode_io.cpp
    src/nl_to_code.cpp
    # NOTE: src/vm.cpp, src/vm_builtins.cpp, src/ffi.cpp, src/project.cpp,
    # src/voice.cpp are EXCLUDED because they need OS syscalls the LSP
    # loop doesn't actually use. If a future LSP feature needs them,
    # add back with #ifdef FOR_WASM guards.
)

# Static library so the WASM build links cleanly
add_library(alphabet_lsp_wasm STATIC ${LIB_SOURCES})

target_compile_options(alphabet_lsp_wasm PRIVATE
    -DFOR_WASM=1
    -s USE_PTHREADS=0
    -s ENVIRONMENT=node
    -s SINGLE_FILE=0
)

# LSP entry executable
add_executable(alphabet_lsp src/wasm_lsp_main.cpp)
target_link_libraries(alphabet_lsp alphabet_lsp_wasm)

# Linker flags for the Node.js target
# - ENVIRONMENT=node: emit Node-friendly WASI bindings
# - STANDALONE_WASM=0 + NODERAWFS=1: use Node fs for filesystem
# - EXPORTED_FUNCTIONS only includes _alphabet_lsp_main (host calls this)
# - ALLOW_MEMORY_GROWTH: LSP needs to grow for large projects
target_link_options(alphabet_lsp PRIVATE
    "-s EXPORTED_FUNCTIONS=['_alphabet_lsp_main','_malloc','_free']"
    "-s EXPORTED_RUNTIME_METHODS=['ccall','cwrap','UTF8ToString','stringToUTF8','lengthBytesUTF8','getValue','setValue']"
    "-s ALLOW_MEMORY_GROWTH=1"
    "-s MAXIMUM_MEMORY=2gb"
    "-s MODULARIZE=1"
    "-s EXPORT_NAME='AlphabetLspModule'"
    "-s WASM=1"
    "-s NO_EXIT_RUNTIME=1"
    "-s ENVIRONMENT=node"
    "-s NODERAWFS=1"
    "-s ASSERTIONS=0"
    "-O2"
)
